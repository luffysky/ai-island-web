### 基本 RAG（3 行 code）

📄 **寫進程式檔**：`rag.py`

```text
from llama_index.core import VectorStoreIndex, SimpleDirectoryReader

# === 1. 載入 ./data 內所有檔案 ===
documents = SimpleDirectoryReader("./data").load_data()

# === 2. 建索引（embedding + 存向量 DB）===
index = VectorStoreIndex.from_documents(documents)

# === 3. 查詢 ===
query_engine = index.as_query_engine()
response = query_engine.query("請假規定？")
print(response)
```

> 💡 **三行 RAG 完成**——LangChain 同樣事要 30 行。

  
### 進階：自訂 retriever

```text
from llama_index.core.retrievers import VectorIndexRetriever
from llama_index.core.query_engine import RetrieverQueryEngine
from llama_index.core.postprocessor import SimilarityPostprocessor

# === 拿 top 10 ===
retriever = VectorIndexRetriever(
    index=index,
    similarity_top_k=10,
)

# === 過濾相似度 < 0.7 的 ===
postprocessor = SimilarityPostprocessor(similarity_cutoff=0.7)

# === 組合 query engine ===
query_engine = RetrieverQueryEngine(
    retriever=retriever,
    node_postprocessors=[postprocessor]
)
```

### 進階 RAG 模式

  
**Hybrid Search（向量 + 關鍵字）：**

```text
from llama_index.retrievers.bm25 import BM25Retriever
from llama_index.core.retrievers import QueryFusionRetriever

vector_retriever = index.as_retriever(similarity_top_k=10)
bm25_retriever = BM25Retriever.from_defaults(index=index)

retriever = QueryFusionRetriever(
    [vector_retriever, bm25_retriever],
    similarity_top_k=10,
    mode="reciprocal_rerank"   # ⭐ 兩個結果合併排序
)
```

  
**Reranking（精準度 +20-30%）：**

```text
from llama_index.postprocessor.cohere_rerank import CohereRerank

reranker = CohereRerank(top_n=5)

query_engine = index.as_query_engine(
    node_postprocessors=[reranker]
)
```

  
**Sub-question（拆解複雜問題）：**

```text
from llama_index.core.query_engine import SubQuestionQueryEngine

query_engine = SubQuestionQueryEngine.from_defaults(
    query_engine_tools=[...]
)

response = query_engine.query(
    "比較公司 A 跟 B 的 2024 收入跟員工數"
)

# === 自動拆成 4 個子問題各自搜尋 ===
# 1. 公司 A 2024 收入？
# 2. 公司 A 2024 員工數？
# 3. 公司 B 2024 收入？
# 4. 公司 B 2024 員工數？
```

### Eval（測 RAG 效果）

```text
from llama_index.core.evaluation import FaithfulnessEvaluator, RelevancyEvaluator
from llama_index.llms.openai import OpenAI

eval_llm = OpenAI(model="gpt-4o")

# === 答案是否基於 context（無幻覺）===
faithfulness = FaithfulnessEvaluator(llm=eval_llm)

# === 答案是否相關 ===
relevancy = RelevancyEvaluator(llm=eval_llm)

# === 評估 ===
response = query_engine.query("請假規定？")
faith_result = faithfulness.evaluate_response(response=response)
print(f"Faithfulness: {faith_result.passing}")
```

### 完整實戰：Notion 內部 KB → RAG

```text
from llama_index.core import VectorStoreIndex
from llama_index.readers.notion import NotionPageReader
from llama_index.llms.anthropic import Anthropic
from llama_index.embeddings.openai import OpenAIEmbedding
from llama_index.postprocessor.cohere_rerank import CohereRerank

# === 1. 從 Notion 載入所有 KB ===
notion_token = "secret_xxx"
documents = NotionPageReader(integration_token=notion_token).load_data(
    database_id="xxxxxxx"   # 你的 Notion DB
)
print(f"載入 {len(documents)} 個頁面")

# === 2. 建索引 ===
llm = Anthropic(model="claude-opus-4-7")
embed_model = OpenAIEmbedding(model="text-embedding-3-small")

index = VectorStoreIndex.from_documents(
    documents,
    embed_model=embed_model
)

# === 3. 建 query engine（加 reranker）===
reranker = CohereRerank(top_n=5)

query_engine = index.as_query_engine(
    llm=llm,
    similarity_top_k=10,
    node_postprocessors=[reranker]
)

# === 4. 查詢 ===
response = query_engine.query("公司請假規定是什麼？")
print(response)

# === 印來源 ===
for node in response.source_nodes:
    print(f"來源：{node.metadata.get('title', 'Unknown')}")
    print(f"分數：{node.score}")
```
