### 基本使用

📄 **寫進程式檔**：`generate.ts`

```text
import { generateText } from "ai";
import { openai } from "@ai-sdk/openai";

const { text } = await generateText({
  model: openai("gpt-4o"),
  prompt: "哈囉"
});

console.log(text);
```

  
**換 Anthropic 只要改一行：**

```text
import { generateText } from "ai";
import { anthropic } from "@ai-sdk/anthropic";

const { text } = await generateText({
  model: anthropic("claude-opus-4-7"),   // ⭐ 只改這行
  prompt: "哈囉"
});
```

  
### 串流（前端）

📄 **寫進程式檔**：`app/api/chat/route.ts`

```text
import { streamText } from "ai";
import { anthropic } from "@ai-sdk/anthropic";

export async function POST(req: Request) {
  const { messages } = await req.json();
  
  const result = await streamText({
    model: anthropic("claude-opus-4-7"),
    messages
  });
  
  return result.toDataStreamResponse();
}
```

### 結構化輸出（Zod）

```text
import { generateObject } from "ai";
import { z } from "zod";

const { object } = await generateObject({
  model: openai("gpt-4o"),
  schema: z.object({
    recipe: z.object({
      name: z.string(),
      ingredients: z.array(z.string()),
      steps: z.array(z.string())
    })
  }),
  prompt: "生成義大利麵食譜"
});

// === 完全 type-safe ===
object.recipe.name;           // string
object.recipe.ingredients;    // string[]
object.recipe.steps;          // string[]
```

> 💡 **Zod schema = TypeScript type + runtime 驗證**——LLM 輸出 100% 符合 schema。

  
### Tool Use

```text
import { generateText, tool } from "ai";
import { z } from "zod";

const { text, toolCalls } = await generateText({
  model: anthropic("claude-opus-4-7"),
  tools: {
    getWeather: tool({
      description: "取得城市天氣",
      parameters: z.object({
        city: z.string()
      }),
      execute: async ({ city }) => {
        const data = await fetch(`/weather/${city}`);
        return await data.json();
      }
    })
  },
  prompt: "台北現在幾度？",
  maxSteps: 5    // ⭐ 限制 agent loop 步驟
});
```

### 完整實戰：ChatGPT clone（50 行 code）

📄 **寫進程式檔**：`app/page.tsx`

```text
'use client';
import { useChat } from "@ai-sdk/react";

export default function Home() {
  const { 
    messages, 
    input, 
    handleSubmit, 
    handleInputChange, 
    isLoading, 
    stop 
  } = useChat();
  
  return (
    <div className="max-w-2xl mx-auto p-4">
      <h1 className="text-2xl font-bold mb-4">My ChatGPT</h1>
      
      <div className="space-y-2 mb-4">
        {messages.map(m => (
          <div key={m.id} className={`p-3 rounded ${m.role === 'user' ? 'bg-blue-50' : 'bg-gray-50'}`}>
            <strong>{m.role === 'user' ? '🧑' : '🤖'}</strong>
            <p className="whitespace-pre-wrap">{m.content}</p>
          </div>
        ))}
      </div>
      
      <form onSubmit={handleSubmit} className="flex gap-2">
        <input 
          value={input} 
          onChange={handleInputChange} 
          className="flex-1 border rounded p-2"
          placeholder="輸入訊息..."
        />
        <button type="submit" disabled={isLoading} className="bg-blue-500 text-white px-4 py-2 rounded">
          Send
        </button>
        {isLoading && (
          <button onClick={stop} type="button" className="bg-red-500 text-white px-4 py-2 rounded">
            Stop
          </button>
        )}
      </form>
    </div>
  );
}
```

📄 **寫進程式檔**：`app/api/chat/route.ts`

