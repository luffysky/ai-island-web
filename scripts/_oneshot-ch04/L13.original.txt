### 事件處理

```js
btn.addEventListener("click", e => {
  e.preventDefault();        // 阻止預設行為
  e.stopPropagation();      // 阻止冒泡
  console.log(e.target);    // 觸發元素
  console.log(e.currentTarget); // 綁定元素
});

// 事件委派：在父元素監聽，靠 e.target 判斷
list.addEventListener("click", e => {
  if (e.target.matches("button.delete")) {
    handleDelete(e.target);
  }
});
// 一個 listener 處理所有子按鈕，效能好

// 移除（必須用同一個函式參考）
function onClick(e) { /*...*/ }
btn.addEventListener("click", onClick);
btn.removeEventListener("click", onClick);

// 選項：once / passive / capture
el.addEventListener("click", fn, { once: true }); // 只觸發一次
el.addEventListener("scroll", fn, { passive: true }); // 不會 preventDefault，效能好
```

### 觀察 API

```js
// IntersectionObserver：元素進入視窗時觸發（lazy load 用）
const io = new IntersectionObserver(entries => {
  entries.forEach(e => {
    if (e.isIntersecting) {
      console.log("進視窗了");
    }
  });
}, { threshold: 0.5 });
io.observe(el);

// MutationObserver：監聽 DOM 變化
const mo = new MutationObserver(mutations => {
  mutations.forEach(m => console.log(m.type));
});
mo.observe(el, { childList: true, subtree: true });

// ResizeObserver：監聽大小變化
const ro = new ResizeObserver(entries => {
  entries.forEach(e => console.log(e.contentRect));
});
ro.observe(el);
```
