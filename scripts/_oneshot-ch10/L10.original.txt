### error.tsx：自動錯誤邊界

```tsx
// app/blog/error.tsx
"use client"; // error.tsx 必須是 Client Component

export default function Error({
  error,
  reset
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  return (
    
      
## 發生錯誤

      
{error.message}

      重試
    
  );
}
// 自動 catch /blog/* 內的所有錯誤
```
