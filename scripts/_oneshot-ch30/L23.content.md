### 1. Email 報表（最基本）

```text
import smtplib
from email.message import EmailMessage
import pandas as pd

# === 準備內容 ===
msg = EmailMessage()
msg["Subject"] = f"每日報表 - {datetime.now():%Y-%m-%d}"
msg["From"] = "bot@example.com"
msg["To"] = "boss@example.com"

# === 文字內容（簡單）===
msg.set_content(f"""
今日抓到 {len(df)} 筆商品
平均價格：NT$ {df['price'].mean():,.0f}
最便宜：NT$ {df['price'].min()}
最貴：NT$ {df['price'].max()}
""")

# === 附 CSV 檔 ===
with open("report.csv", "rb") as f:
    msg.add_attachment(
        f.read(),
        maintype="text",
        subtype="csv",
        filename=f"report_{datetime.now():%Y%m%d}.csv"
    )

# === 寄出 ===
with smtplib.SMTP("smtp.gmail.com", 587) as s:
    s.starttls()
    s.login(USER, PASS)
    s.send_message(msg)
```

📄 **寫進程式檔**：`dashboard.py`

```text
import streamlit as st
import pandas as pd
import plotly.express as px

st.set_page_config(page_title="📊 爬蟲報表", layout="wide")
st.title("📊 商品比價儀表板")

# === 載入資料 ===
df = pd.read_parquet("data.parquet")

# === 3 個指標 ===
col1, col2, col3 = st.columns(3)
col1.metric("總商品", f"{len(df):,}")
col2.metric("平均價", f"NT$ {df['price'].mean():,.0f}")
col3.metric("分類數", df['category'].nunique())

# === 分類平均價長條圖 ===
st.subheader("分類平均價")
fig = px.bar(
    df.groupby("category")["price"].mean().reset_index(),
    x="category",
    y="price",
    color="category"
)
st.plotly_chart(fig, use_container_width=True)

# === 互動篩選 ===
st.subheader("商品清單")
category = st.selectbox("選分類", df['category'].unique())
filtered = df[df['category'] == category]
st.dataframe(filtered)
```

### 4. Next.js + Recharts Dashboard（最完整）

📄 **寫進程式檔**：`app/dashboard/page.tsx`

```text
'use client';
import { LineChart, Line, XAxis, YAxis, Tooltip, Legend } from 'recharts';
import useSWR from 'swr';

const fetcher = (url: string) => fetch(url).then(r => r.json());

export default function Dashboard() {
  const { data, error } = useSWR('/api/products/trend', fetcher, {
    refreshInterval: 60_000  // 每分鐘更新
  });
  
  if (error) return <div>載入失敗</div>;
  if (!data) return <div>載入中...</div>;
  
  return (
    <div className="p-8">
      <h1 className="text-2xl font-bold mb-4">商品趨勢</h1>
      
      <LineChart width={800} height={400} data={data}>
        <XAxis dataKey="date" />
        <YAxis />
        <Tooltip />
        <Legend />
        <Line type="monotone" dataKey="shopee" stroke="#f97316" />
        <Line type="monotone" dataKey="momo" stroke="#3b82f6" />
        <Line type="monotone" dataKey="pchome" stroke="#10b981" />
      </LineChart>
    </div>
  );
}
```

  
### 5. REST API（FastAPI）

```text
from fastapi import FastAPI
from sqlalchemy.orm import Session

app = FastAPI()

@app.get("/products")
def get_products(limit: int = 100, category: str | None = None):
    query = "SELECT * FROM products"
    if category:
        query += f" WHERE category = '{category}'"
    query += f" LIMIT {limit}"
    
    return db.query(query)

@app.get("/stats")
def stats():
    return {
        "total": db.query_one("SELECT COUNT(*) FROM products"),
        "avg_price": db.query_one("SELECT AVG(price) FROM products"),
        "categories": db.query("SELECT category, COUNT(*) FROM products GROUP BY category"),
    }

@app.get("/products/{product_id}/history")
def price_history(product_id: int):
    return db.query("""
        SELECT date, price 
        FROM prices 
        WHERE product_id = $1 
        ORDER BY date
    """, product_id)
```

```text
from weasyprint import HTML
from jinja2 import Template

template = Template("""
<html>
<head><style>
body { font-family: 'Noto Sans TC', sans-serif; }
table { border-collapse: collapse; width: 100%; }
th, td { border: 1px solid #ddd; padding: 8px; text-align: left; }
th { background: #f0f0f0; }
</style></head>
<body>
  <h1>{{ title }}</h1>
  <p>產生時間：{{ date }}</p>
  <table>
    <tr>{% for col in columns %}<th>{{ col }}</th>{% endfor %}</tr>
    {% for row in rows %}
    <tr>{% for cell in row %}<td>{{ cell }}</td>{% endfor %}</tr>
    {% endfor %}
  </table>
</body>
</html>
""")

html = template.render(
    title="每日商品報表",
    date=datetime.now(),
    columns=df.columns,
    rows=df.values.tolist()
)

HTML(string=html).write_pdf("report.pdf")
```

### 4 大心法

```text
1. ⭐ 客戶看不到 = 沒做
   → 爬蟲跑得再厲害、不交付就 0
   → 永遠想：客戶怎麼看到？

2. ⭐ 越漂亮 = 越好賣
   → CSV vs Dashboard、實際工作量差不大
   → 但客戶感受差 10x

3. ⭐ 給 Dashboard > 給 CSV
   → 報價可 5-10x
   → 客戶愛、續約率高

4. ⭐ 自動更新 = 月費合約
   → 一次性 NT$ 30K vs 月費 NT$ 5K × 12 = NT$ 60K
   → 月費商業模式好
```
