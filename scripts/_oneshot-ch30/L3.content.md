### 第一個 Rust 爬蟲

📄 **寫進程式檔**：`src/main.rs`

```text
use reqwest;
use scraper::{Html, Selector};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // === 1. 抓 HTML ===
    let body = reqwest::get("https://books.toscrape.com")
        .await?
        .text()
        .await?;
    
    // === 2. 解析 ===
    let document = Html::parse_document(&body);
    let book_sel = Selector::parse("article.product_pod").unwrap();
    let title_sel = Selector::parse("h3 a").unwrap();
    let price_sel = Selector::parse(".price_color").unwrap();
    
    // === 3. 迭代每本書 ===
    for book in document.select(&book_sel) {
        let title = book.select(&title_sel)
            .next()
            .and_then(|e| e.value().attr("title"))
            .unwrap_or("");
        
        let price = book.select(&price_sel)
            .next()
            .map(|e| e.text().collect::<String>())
            .unwrap_or_default();
        
        println!("{} | {}", title, price);
    }
    
    Ok(())
}
```

### 並發爬多頁（Tokio）

```text
use futures::stream::{self, StreamExt};
use reqwest::Client;

#[tokio::main]
async fn main() {
    let client = Client::new();
    
    // === 生 100 個 URL ===
    let urls: Vec<String> = (1..=100)
        .map(|i| format!("https://example.com/page/{}", i))
        .collect();
    
    // === 並發 10、處理 100 URL（buffer_unordered 控制並發）===
    let bodies: Vec<_> = stream::iter(urls)
        .map(|url| {
            let client = client.clone();
            async move {
                let resp = client.get(&url).send().await;
                resp?.text().await
            }
        })
        .buffer_unordered(10)                  // 同時最多 10 個 future 跑
        .collect()
        .await;
    
    println!("Fetched {} pages", bodies.len());
}
```

> 💡 **`buffer_unordered(10)`** = Rust 版的 `asyncio.Semaphore(10)` / `pLimit(10)`、控制並發數。

```text

```

**這行做什麼？** 裝 chromiumoxide、Rust 控 Chrome 的套件（相當於 Puppeteer）、用 tokio runtime。

```text
use chromiumoxide::{Browser, BrowserConfig};
use futures::StreamExt;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // === 啟動瀏覽器 ===
    let (browser, mut handler) = Browser::launch(
        BrowserConfig::builder().build()?
    ).await?;
    
    // === handler 在背景跑（處理瀏覽器事件）===
    let _handle = tokio::task::spawn(async move {
        while let Some(_) = handler.next().await {}
    });
    
    // === 開頁 ===
    let page = browser.new_page("https://example.com").await?;
    let title = page.get_title().await?;
    
    println!("Title: {:?}", title);
    
    Ok(())
}
```
