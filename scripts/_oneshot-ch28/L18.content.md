### 完整一條龍範例（電商商品分析）

```text
import pandas as pd
import requests
from bs4 import BeautifulSoup

def scrape_products(pages=10):
    """爬多頁商品、回傳 list of dict"""
    products = []
    for p in range(1, pages + 1):
        res = requests.get(f'https://example.com/products?page={p}', timeout=10)
        soup = BeautifulSoup(res.text, 'lxml')
        
        for card in soup.select('div.product'):
            products.append({
                'name': card.select_one('.name').get_text(strip=True),
                'price': card.select_one('.price').get_text(strip=True),
                'category': card.select_one('.category').get_text(strip=True),
            })
    return products

# === 1. 爬 ===
df = pd.DataFrame(scrape_products())
print(f'爬到 {len(df)} 筆商品')

# === 2. 清理（爬下來是字串「NT$ 1,299」、要轉 float）===
df['price'] = (
    df['price']
    .str.replace('NT$', '', regex=False)
    .str.replace(',', '', regex=False)
    .astype(float)
)

# === 3. 分析（Pandas groupby、ch27 L4 教過）===
print(df.groupby('category')['price'].agg(['mean', 'median', 'count']))

# === 4. Top 10 最貴 ===
top10 = df.nlargest(10, 'price')
print(top10[['name', 'price', 'category']])

# === 5. 存 CSV + Parquet（兩格式給不同需求）===
df.to_csv('products.csv', index=False, encoding='utf-8-sig')
df.to_parquet('products.parquet')             # 給之後分析用、快 + 小
```

import pandas as pd
from datetime import datetime

# === 加 timestamp ===
df['date'] = pd.Timestamp.today().date()

# === 載入歷史 + append ===
try:
    history = pd.read_parquet('history.parquet')
    combined = pd.concat([history, df], ignore_index=True)
except FileNotFoundError:
    combined = df

combined.to_parquet('history.parquet')

# === 分析價格變化（ch27 L6 時序教過）===
combined['date'] = pd.to_datetime(combined['date'])

# 看某商品歷史價格
product_history = combined[combined['name'] == 'iPhone 17']
product_history.set_index('date')['price'].plot(title='iPhone 17 價格變化')

### Streamlit 一鍵變 dashboard（ch27 L12 教過、這裡整合）

```text
# dashboard.py
import streamlit as st
import pandas as pd

st.title('📊 商品分析 Dashboard')

@st.cache_data
def load():
    return pd.read_parquet('history.parquet')

df = load()

# === Sidebar 篩選 ===
category = st.sidebar.selectbox('類別', df['category'].unique())
filtered = df[df['category'] == category]

# === KPI ===
col1, col2, col3 = st.columns(3)
col1.metric('商品數', len(filtered))
col2.metric('均價', f"NT$ {filtered['price'].mean():,.0f}")
col3.metric('最高價', f"NT$ {filtered['price'].max():,.0f}")

# === 圖表 ===
st.bar_chart(filtered.groupby('category')['price'].mean())

# === 表格 ===
st.dataframe(filtered)
```
