### 第一個 Spider

```python
# myspider/spiders/books.py
import scrapy

class BooksSpider(scrapy.Spider):
    name = 'books'
    start_urls = ['http://books.toscrape.com']

    def parse(self, response):
        # 抓書名 + 價格
        for book in response.css('article.product_pod'):
            yield {
                'title': book.css('h3 a::attr(title)').get(),
                'price': book.css('.price_color::text').get(),
                'stock': book.css('.availability::text').get().strip(),
            }

        # 找下一頁、繼續爬
        next_page = response.css('li.next a::attr(href)').get()
        if next_page:
            yield response.follow(next_page, callback=self.parse)
```


### 執行

### Item Pipeline（資料處理 / 儲存）

```python
# pipelines.py
import pymongo

class MongoDBPipeline:
    def open_spider(self, spider):
        self.client = pymongo.MongoClient('mongodb://localhost:27017')
        self.db = self.client['scrapy']

    def process_item(self, item, spider):
        # 去重、清洗、儲存
        if self.db.books.find_one({'title': item['title']}):
            raise scrapy.exceptions.DropItem('重複')

        # 清洗價格
        item['price'] = float(item['price'].replace('£', ''))

        self.db.books.insert_one(dict(item))
        return item

    def close_spider(self, spider):
        self.client.close()

# settings.py 啟用
ITEM_PIPELINES = {
    'myspider.pipelines.MongoDBPipeline': 300,
}
```
