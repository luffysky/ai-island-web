function Form() {
    const handleSubmit = (e: React.FormEvent) => {
        e.preventDefault();        // 阻止預設行為（不跳轉頁面）
        console.log("submitted");
    };

    const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
        console.log(e.target.value);
    };

    const handleClick = (e: React.MouseEvent) => {
        console.log(e.clientX, e.clientY);
        e.stopPropagation();    // 阻止冒泡到父
    };

    return (
        <form onSubmit={handleSubmit}>
            <input onChange={handleChange} />
            <button type="submit" onClick={handleClick}>Submit</button>
        </form>
    );
}

// Controlled input（推薦）
function LoginForm() {
    const [email, setEmail] = useState("");
    const [password, setPassword] = useState("");

    const handleSubmit = async (e: React.FormEvent) => {
        e.preventDefault();

        try {
            await login(email, password);
            // 成功
        } catch (err) {
            // 失敗
        }
    };

    return (
        <form onSubmit={handleSubmit}>
            <input
                type="email"
                value={email}
                onChange={e => setEmail(e.target.value)}
                required
            />
            <input
                type="password"
                value={password}
                onChange={e => setPassword(e.target.value)}
                required
            />
            <button type="submit">Login</button>
        </form>
    );
}

// Controlled：value 跟 state 同步
// Uncontrolled：用 ref（少用）

// 多個輸入共用一個 handler
const [form, setForm] = useState({
    name: "",
    email: "",
    password: ""
});

const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target;
    setForm(prev => ({ ...prev, [name]: value }));
};

// preventDefault：阻止瀏覽器預設行為
function Link() {
    const handleClick = (e: React.MouseEvent) => {
        e.preventDefault();   // 不跳轉頁面
        // 自己處理
    };

    return <a href="/page" onClick={handleClick}>Click</a>;
}

// stopPropagation：阻止冒泡到父層
function Parent() {
    return (
        <div onClick={() => console.log("parent")}>
            <button onClick={(e) => {
                e.stopPropagation();   // parent 的 onClick 不會跑
                console.log("button");
            }}>
                Click
            </button>
        </div>
    );
}

function Button() {
    const handleClick = () => {
        console.log("clicked");
    };

    // 事件處理：傳 function reference、不要直接呼叫
    return (
        <div>
            <button onClick={handleClick}>Click（傳 reference）</button>
            <button onClick={() => console.log("inline clicked")}>
                Click（inline 箭頭函式）
            </button>
        </div>
    );
}

// ❌ <button onClick={handleClick()}>  // 立刻執行、不是綁事件
// ✓ <button onClick={handleClick}>     // 傳 reference、被點才跑
