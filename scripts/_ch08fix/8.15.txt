// ❌ 手動驗證：又臭又長
function LoginForm() {
    const [email, setEmail] = useState("");
    const [password, setPassword] = useState("");
    const [emailError, setEmailError] = useState("");
    const [passwordError, setPasswordError] = useState("");
    const [submitting, setSubmitting] = useState(false);

    const handleSubmit = async (e: React.FormEvent) => {
        e.preventDefault();

        let valid = true;
        if (!email.includes("@")) {
            setEmailError("Invalid email");
            valid = false;
        } else setEmailError("");

        if (password.length < 8) {
            setPasswordError("密碼至少 8 碼");
            valid = false;
        } else setPasswordError("");

        if (!valid) return;

        setSubmitting(true);
        await login(email, password);
        setSubmitting(false);
    };

    return (
        <form onSubmit={handleSubmit}>
            <input value={email} onChange={(e) => setEmail(e.target.value)} />
            {emailError && <span className="error">{emailError}</span>}
            <input type="password" value={password} onChange={(e) => setPassword(e.target.value)} />
            {passwordError && <span className="error">{passwordError}</span>}
            <button disabled={submitting}>{submitting ? "Loading..." : "Login"}</button>
        </form>
    );
}

// ✓ React Hook Form + Zod：短、type-safe、性能好
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";

const LoginSchema = z.object({
    email: z.string().email(),
    password: z.string().min(8)
});
type LoginInput = z.infer<typeof LoginSchema>;

function LoginFormRHF() {
    const {
        register,
        handleSubmit,
        formState: { errors, isSubmitting }
    } = useForm<LoginInput>({
        resolver: zodResolver(LoginSchema)
    });

    const onSubmit = async (data: LoginInput) => {
        await login(data.email, data.password);
    };

    return (
        <form onSubmit={handleSubmit(onSubmit)}>
            <label>
                Email
                <input {...register("email")} />
                {errors.email && <span>{errors.email.message}</span>}
            </label>

            <label>
                Password
                <input type="password" {...register("password")} />
                {errors.password && <span>{errors.password.message}</span>}
            </label>

            <button disabled={isSubmitting}>
                {isSubmitting ? "Loading..." : "Login"}
            </button>
        </form>
    );
}

import { useForm, useFieldArray } from "react-hook-form";

function ContactsForm() {
    const { register, control, handleSubmit } = useForm({
        defaultValues: {
            name: "",
            contacts: [{ type: "email", value: "" }]
        }
    });

    const { fields, append, remove } = useFieldArray({
        control,
        name: "contacts"
    });

    return (
        <form onSubmit={handleSubmit(onSubmit)}>
            <input {...register("name")} />

            {fields.map((field, index) => (
                <div key={field.id}>
                    <select {...register(`contacts.${index}.type`)}>
                        <option value="email">Email</option>
                        <option value="phone">Phone</option>
                    </select>
                    <input {...register(`contacts.${index}.value`)} />
                    <button type="button" onClick={() => remove(index)}>Remove</button>
                </div>
            ))}

            <button type="button" onClick={() => append({ type: "email", value: "" })}>
                Add Contact
            </button>

            <button type="submit">Save</button>
        </form>
    );
}

// app/actions.ts
"use server";

import { z } from "zod";

const Schema = z.object({
    email: z.string().email(),
    password: z.string().min(8)
});

export async function loginAction(formData: FormData) {
    const parsed = Schema.safeParse({
        email: formData.get("email"),
        password: formData.get("password")
    });

    if (!parsed.success) {
        return { error: parsed.error.flatten() };
    }

    await db.login(parsed.data.email, parsed.data.password);
    redirect("/dashboard");
}

// Component
function LoginForm() {
    return (
        <form action={loginAction}>
            <input name="email" type="email" />
            <input name="password" type="password" />
            <button type="submit">Login</button>
        </form>
    );
}
