### Pipeline：3 行用 AI

```text
from transformers import pipeline

# 情感分析
classifier = pipeline("sentiment-analysis")
result = classifier("I love this product!")
# [{'label': 'POSITIVE', 'score': 0.9998}]

# 翻譯
translator = pipeline("translation_en_to_zh")
translator("Hello, how are you?")

# 摘要
summarizer = pipeline("summarization")
summary = summarizer(long_text, max_length=50)

# 問答
qa = pipeline("question-answering")
qa(question="Who created Python?",
   context="Python was created by Guido van Rossum in 1991.")

# 文字生成
generator = pipeline("text-generation", model="gpt2")
generator("The future of AI is", max_length=50)

# Whisper 語音轉文字
transcriber = pipeline("automatic-speech-recognition",
                       model="openai/whisper-base")
transcriber("audio.mp3")

# 圖像分類
classifier = pipeline("image-classification")
classifier("dog.jpg")

# Stable Diffusion
from diffusers import StableDiffusionPipeline
pipe = StableDiffusionPipeline.from_pretrained(
    "runwayml/stable-diffusion-v1-5"
)
image = pipe("a cat astronaut in space").images[0]
image.save("cat.png")
```

### 底層 API

```text
from transformers import AutoTokenizer, AutoModelForSequenceClassification
import torch

model_name = "bert-base-uncased"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForSequenceClassification.from_pretrained(model_name)

# Tokenize
text = "I love this product"
inputs = tokenizer(text, return_tensors="pt",
                   padding=True, truncation=True)

# 推論
with torch.no_grad():
    outputs = model(**inputs)
    logits = outputs.logits
    probs = torch.softmax(logits, dim=-1)
    predicted = torch.argmax(probs, dim=-1)

# 對應 task：
# AutoModelForSequenceClassification（分類）
# AutoModelForCausalLM（生成）
# AutoModelForQuestionAnswering（QA）
# AutoModelForTokenClassification（NER）
```

  
### Fine-tune：客製化

```text
from transformers import (
    AutoTokenizer, AutoModelForSequenceClassification,
    Trainer, TrainingArguments
)
from datasets import Dataset

# 1. 資料
data = {
    "text": ["這產品很棒", "服務很差", "普通", "超讚", "爛透"],
    "label": [1, 0, 1, 1, 0]
}
dataset = Dataset.from_dict(data)
dataset = dataset.train_test_split(test_size=0.2)

# 2. Tokenize
model_name = "bert-base-chinese"
tokenizer = AutoTokenizer.from_pretrained(model_name)

def tokenize(batch):
    return tokenizer(batch["text"], truncation=True, padding=True)

tokenized = dataset.map(tokenize, batched=True)

# 3. Model
model = AutoModelForSequenceClassification.from_pretrained(
    model_name, num_labels=2
)

# 4. Training args
training_args = TrainingArguments(
    output_dir="./results",
    num_train_epochs=3,
    per_device_train_batch_size=16,
    learning_rate=2e-5,
    eval_strategy="epoch",
    save_strategy="epoch",
    load_best_model_at_end=True,
)

# 5. Trainer
trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=tokenized["train"],
    eval_dataset=tokenized["test"],
    tokenizer=tokenizer,
)
trainer.train()

# 6. 存
trainer.save_model("./my-model")
tokenizer.save_pretrained("./my-model")
```

  
### LoRA / PEFT（高效 fine-tune）

```text
# 完整 fine-tune Llama 70B = 100GB GPU
# LoRA：只 fine-tune 0.1% 參數、結果幾乎一樣

from peft import LoraConfig, get_peft_model

model = AutoModelForCausalLM.from_pretrained("meta-llama/Llama-3-8B")

lora_config = LoraConfig(
    r=8,
    lora_alpha=16,
    target_modules=["q_proj", "v_proj"],
    lora_dropout=0.05,
    bias="none",
    task_type="CAUSAL_LM"
)

model = get_peft_model(model, lora_config)
model.print_trainable_parameters()
# trainable params: 4M / 8B（0.05%）

# 2026 主流：LoRA + QLoRA（量化 + LoRA）
# Single GPU fine-tune Llama 70B
```
