### DB Schema（Drizzle）

📄 **寫進程式檔**：`src/db/schema.ts`

```text
import { pgTable, uuid, text, decimal, timestamp, integer } from 'drizzle-orm/pg-core';

// === 商品（使用者訂閱的）===
export const products = pgTable('products', {
  id: uuid('id').defaultRandom().primaryKey(),
  user_id: uuid('user_id').notNull(),
  keyword: text('keyword').notNull(),
  target_price: decimal('target_price').notNull(),
  created_at: timestamp('created_at').defaultNow(),
});

// === 每天爬到的價格（歷史走勢）===
export const prices = pgTable('prices', {
  id: uuid('id').defaultRandom().primaryKey(),
  product_id: uuid('product_id').references(() => products.id),
  site: text('site').notNull(),                // 'shopee' / 'pchome' / 'momo'
  title: text('title').notNull(),
  price: decimal('price').notNull(),
  url: text('url').notNull(),
  scraped_at: timestamp('scraped_at').defaultNow(),
});
```

  
### 爬蟲 1：蝦皮（Playwright + 攔 API）

📄 **寫進程式檔**：`src/scrapers/shopee.ts`

```text
import { chromium } from 'playwright';

export async function scrapeShopee(keyword: string) {
  const browser = await chromium.launch();
  const page = await browser.newPage();
  
  const apiData: any[] = [];
  
  // === 攔 API（神技、跳過解 HTML）===
  page.on('response', async (res) => {
    if (res.url().includes('/api/v4/search')) {
      try {
        const data = await res.json();
        if (data.items) apiData.push(...data.items);
      } catch {}
    }
  });
  
  await page.goto(
    `https://shopee.tw/search?keyword=${encodeURIComponent(keyword)}`,
    { waitUntil: 'networkidle' }
  );
  
  await browser.close();
  
  // === 整理前 20 個 ===
  return apiData.slice(0, 20).map(item => ({
    title: item.name,
    price: item.price / 100000,                // 蝦皮價格除 100000
    url: `https://shopee.tw/product/${item.shopid}/${item.itemid}`,
  }));
}
```

### Cron 主程式（每天 6am 跑）

📄 **寫進程式檔**：`src/cron/scrape-all.ts`

```text
import { db } from '../db';
import { products, prices } from '../db/schema';
import { scrapeShopee } from '../scrapers/shopee';
import { scrapePChome } from '../scrapers/pchome';
import { checkAndNotify } from './notify';

export async function main() {
  // === 1. 拿所有訂閱的商品 ===
  const allProducts = await db.select().from(products);
  console.log(`今天要爬 ${allProducts.length} 個商品`);
  
  // === 2. 每個商品爬 3 個網站（並行）===
  for (const p of allProducts) {
    try {
      const [shopee, pchome] = await Promise.all([
        scrapeShopee(p.keyword),
        scrapePChome(p.keyword),
      ]);
      
      // === 3. 寫入 DB ===
      await db.insert(prices).values([
        ...shopee.map(i => ({ ...i, site: 'shopee', product_id: p.id })),
        ...pchome.map(i => ({ ...i, site: 'pchome', product_id: p.id })),
      ]);
      
      // === 4. 檢查降價、推 LINE ===
      await checkAndNotify(p.id);
      
      console.log(`✓ ${p.keyword}: shopee ${shopee.length} + pchome ${pchome.length}`);
    } catch (err) {
      console.error(`✗ ${p.keyword}:`, err);
    }
  }
}
```

  
### LINE 通知（降價時推播）

📄 **寫進程式檔**：`src/cron/notify.ts`

```text
import { db } from '../db';
import { prices, subscriptions } from '../db/schema';
import { eq, and, gte } from 'drizzle-orm';

export async function checkAndNotify(productId: string) {
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  // === 拿今天所有價格 ===
  const todayPrices = await db
    .select()
    .from(prices)
    .where(and(
      eq(prices.product_id, productId),
      gte(prices.scraped_at, today)
    ));
  
  if (todayPrices.length === 0) return;
  
  // === 找最低價 ===
  const minPrice = Math.min(...todayPrices.map(p => Number(p.price)));
  const minItem = todayPrices.find(p => Number(p.price) === minPrice)!;
  
  // === 拿所有訂閱、看誰要通知 ===
  const subs = await db
    .select()
    .from(subscriptions)
    .where(eq(subscriptions.product_id, productId));
  
  for (const sub of subs) {
    if (minPrice < Number(sub.target_price)) {
      await sendLineMessage(sub.line_user_id, 
        `💰 降價了！\n${minItem.title}\nNT$ ${minPrice}\n${minItem.url}`
      );
    }
  }
}

async function sendLineMessage(userId: string, text: string) {
  await fetch('https://api.line.me/v2/bot/message/push', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.LINE_TOKEN}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      to: userId,
      messages: [{ type: 'text', text }],
    })
  });
}
```

  
### Hono API（給前端用）

📄 **寫進程式檔**：`src/api/index.ts`

```text
import { Hono } from 'hono';
import { db } from '../db';
import { prices, products } from '../db/schema';
import { eq } from 'drizzle-orm';

const app = new Hono();

// === 拿某商品歷史走勢 ===
app.get('/api/prices/:productId', async (c) => {
  const productId = c.req.param('productId');
  
  const data = await db
    .select()
    .from(prices)
    .where(eq(prices.product_id, productId))
    .orderBy(prices.scraped_at);
  
  return c.json(data);
});

// === 新增訂閱 ===
app.post('/api/products', async (c) => {
  const { user_id, keyword, target_price } = await c.req.json();
  
  const [product] = await db
    .insert(products)
    .values({ user_id, keyword, target_price })
    .returning();
  
  return c.json(product);
});

export default {
  port: 3000,
  fetch: app.fetch,
};
```

'use client';
import { LineChart, Line, XAxis, YAxis, Tooltip } from 'recharts';
import { useEffect, useState } from 'react';

export default function ProductPage({ params }: { params: { id: string } }) {
  const [data, setData] = useState([]);
  
  useEffect(() => {
    fetch(`/api/prices/${params.id}`)
      .then(r => r.json())
      .then(prices => {
        // === 把資料按日期聚合：每日各站最低價 ===
        const byDate = groupByDate(prices);
        setData(byDate);
      });
  }, [params.id]);
  
  return (
    <div className="p-8">
      <h1 className="text-2xl font-bold mb-4">價格走勢</h1>
      <LineChart width={800} height={400} data={data}>
        <XAxis dataKey="date" />
        <YAxis />
        <Tooltip />
        <Line type="monotone" dataKey="shopee" stroke="#f97316" />
        <Line type="monotone" dataKey="pchome" stroke="#3b82f6" />
        <Line type="monotone" dataKey="momo" stroke="#10b981" />

}
